
// Entry point must be a voltage source "function"
AC_Source("my_voltage_source", 5V, 50Hz, 25ohm, 0rad) {
    
    // Each sequential component is connected in series unless Parrallel conenction specified
    Capacitor("C1", 0.0001F);
    
    // Parrallel connections are made by creating a Parrallel "function"
    Parrallel("Combination_1", Null) {
        Resistor("R1", 100ohm);
        Resistor("R2", 50ohm);
        Resistor("R3", 25ohm);
    }

    // Parrallel funcions can be in Parrallel with another Parrallel function
    Parrallel("Combination_2", Combination_1) {
        Resistor("R1", 100ohm);
    }

    Inductor("L1", 0.001H);
}
